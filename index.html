    import random
    from tkinter import *


    def rand():
        hexSystemStr = "0123456789abcdef"
        listHex = ["#"]
        while len(listHex) <= 6:
            a = random.choice(hexSystemStr)
            listHex.append(a)

            strHex = ''.join(listHex)
            print(strHex)
            txt.insert(END, strHex)

            return strHex

    def clear():
        txt.delete(0.0, END)

        root = Tk()
        root.title("HEX-color")
        root.geometry("400x130")
        root.resizable(width=False, 
        height=False)

        txt = Text(width=15, height=1, 
        font="Arial 26")
        txt.pack()

        btn = Button(text="Generate", 
        width=16, height=3, command=rand, 
        font=20, state=NORMAL, bg="#ADFF2F")
        btn.pack(side=LEFT)

        btn = Button(text="Clear", 
        width=16, height=3, command=clear, 
        bg="#CD5C5C", font=20)

        btn.pack(side=RIGHT)


        root.mainloop()
